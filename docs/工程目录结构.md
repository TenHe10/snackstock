SnackStock/
├── main.py                 # 程序入口，启动主界面
├── config.py               # 配置文件（如数据库路径、预警阈值）
├── requirements.txt        # 依赖包列表
│
├── database/               # 数据库相关
│   ├── inventory.db        # 默认 SQLite 主数据库
│   ├── schema.sql          # 数据库建表语句
│   ├── .selected_db_path   # 最近一次在 UI 选择的数据库路径
│   └── archives/           # 按月归档的历史流水分文件
│       └── stock_logs_YYYY_MM.db
│
├── src/                    # 核心源代码
│   ├── db_manager.py       # 数据库访问、库存计算、流水归档
│   ├── scanner_handler.py  # 扫码枪输入解析逻辑（可复用）
│   ├── logic/              # 业务逻辑层
│   │   ├── inbound.py      # 入库逻辑
│   │   ├── outbound.py     # 出库与收银逻辑
│   │   └── report.py       # 报表生成逻辑
│   │
│   └── gui/                # 图形界面层
│       └── main_window.py  # 主窗口布局与交互
│
├── docs/                   # 项目文档
├── build_windows.bat       # Windows 打包脚本
└── SnackStock.spec         # PyInstaller 固定构建配置
