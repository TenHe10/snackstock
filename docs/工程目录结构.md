SnackStock/
├── main.py                 # 程序入口，启动主界面
├── config.py               # 配置文件（如数据库路径、预警阈值）
├── requirements.txt        # 依赖包列表（PyQt6, sqlite3等）
│
├── database/               # 数据库相关
│   ├── inventory.db        # 实际的 SQLite 数据库文件
│   └── schema.sql          # 数据库建表语句备份
│
├── src/                    # 核心源代码
│   ├── db_manager.py       # 数据库增删改查逻辑
│   ├── scanner_handler.py  # 扫码枪输入解析逻辑
│   ├── logic/              # 业务逻辑层
│   │   ├── inbound.py      # 入库逻辑
│   │   ├── outbound.py     # 出库与收银逻辑
│   │   └── report.py       # 报表生成逻辑
│   │
│   └── gui/                # 图形界面层
│       ├── main_window.py  # 主窗口布局
│       ├── widgets/        # 自定义小部件（如库存预警弹窗）
│       └── style.qss       # 界面样式表（美化 UI）
│
├── assets/                 # 静态资源
│   ├── icons/              # 图标（如零食图标、设置图标）
│   └── sounds/             # 扫码成功/失败的提示音
│
├── logs/                   # 日志文件（记录系统报错或操作流水）
│   └── app.log
│
└── utils/                  # 通用工具类
    ├── helpers.py          # 时间格式化、金额计算等小工具
    └── backup_tool.py      # 数据库自动备份工具
